
// READ MPU DATA
#include <BLEDevice.h>
#include <BLEServer.h>
#include <BLEUtils.h>
#include <BLE2902.h>
#include <Wire.h>
#include <math.h>




// BLE
BLECharacteristic *pCharacteristic;
bool deviceConnected = false;

unsigned long lastBleMs = 0;
const unsigned long bleIntervalMs = 100;  // 10 Hz

// UUIDs
#define SERVICE_UUID "12345678-1234-1234-1234-1234567890ab"
#define CHARACTERISTIC_UUID "abcdefab-1234-1234-1234-abcdefabcdef"

class MyServerCallbacks : public BLEServerCallbacks {
  void onConnect(BLEServer *pServer) {
    deviceConnected = true;
    Serial.println("BLE connected");
  }

  void onDisconnect(BLEServer *pServer) {
    deviceConnected = false;
    Serial.println("BLE disconnected");
    BLEDevice::startAdvertising();
    Serial.println("BLE advertising restarted");
  }
};




// MPU
#define SDA_PIN 8
#define SCL_PIN 9
#define MPU9250_ADDR 0x68

// REGISTERS
#define WHO_AM_I_REG 0x75
#define PWR_MGMT_1 0x6B
#define SMPLRT_DIV 0x19
#define CONFIG_REG 0x1A
#define GYRO_CONFIG 0x1B
#define ACCEL_CONFIG 0x1C
#define ACCEL_CONFIG2 0x1D
#define ACCEL_XOUT_H 0x3B




// HELPERS
void writeReg(uint8_t reg, uint8_t value) {
  Wire.beginTransmission(MPU9250_ADDR);
  Wire.write(reg);
  Wire.write(value);
  Wire.endTransmission();
}

bool readRegs(uint8_t reg, uint8_t *buf, size_t len) {
  Wire.beginTransmission(MPU9250_ADDR);
  Wire.write(reg);
  if (Wire.endTransmission(false) != 0) {
    return false;
  }

  size_t n = Wire.requestFrom(MPU9250_ADDR, (uint8_t)len);
  if (n != len) {
    return false;
  }

  for (size_t i = 0; i < len; i++) {
    buf[i] = Wire.read();
  }
  return true;
}

int16_t makeInt16(uint8_t hi, uint8_t lo) {
  return (int16_t)((hi << 8) | lo);
}




// SMOOTHING
float acc_mag_f = 0;
float gyro_mag_f = 0;
const float alpha = 0.1;




// CALIBRATE
float gx_bias = 0.0f;
float gy_bias = 0.0f;
float gz_bias = 0.0f;

bool readMPURaw(
  int16_t &ax_raw, int16_t &ay_raw, int16_t &az_raw,
  int16_t &temp_raw,
  int16_t &gx_raw, int16_t &gy_raw, int16_t &gz_raw) {
  uint8_t raw[14];

  if (!readRegs(ACCEL_XOUT_H, raw, 14)) {
    return false;
  }

  ax_raw = makeInt16(raw[0], raw[1]);
  ay_raw = makeInt16(raw[2], raw[3]);
  az_raw = makeInt16(raw[4], raw[5]);
  temp_raw = makeInt16(raw[6], raw[7]);
  gx_raw = makeInt16(raw[8], raw[9]);
  gy_raw = makeInt16(raw[10], raw[11]);
  gz_raw = makeInt16(raw[12], raw[13]);

  return true;
}

void calibrateGyro(int n_samples = 500, int delay_ms = 5) {
  Serial.println("Keep sensor perfectly still: gyro calibration starting...");

  float sx = 0.0f;
  float sy = 0.0f;
  float sz = 0.0f;

  int ok = 0;

  for (int i = 0; i < n_samples; i++) {
    int16_t ax_raw, ay_raw, az_raw, temp_raw, gx_raw, gy_raw, gz_raw;

    if (readMPURaw(ax_raw, ay_raw, az_raw, temp_raw, gx_raw, gy_raw, gz_raw)) {
      sx += gx_raw / 131.0f;
      sy += gy_raw / 131.0f;
      sz += gz_raw / 131.0f;
      ok++;
    }

    delay(delay_ms);
  }

  if (ok > 0) {
    gx_bias = sx / ok;
    gy_bias = sy / ok;
    gz_bias = sz / ok;
  }

  Serial.print("Gyro bias dps: ");
  Serial.print(gx_bias, 6);
  Serial.print(", ");
  Serial.print(gy_bias, 6);
  Serial.print(", ");
  Serial.println(gz_bias, 6);
}




// TILT
const float tilt_thresh_deg = 20.0f;




// MAIN
void setup() {
  Serial.begin(115200);
  delay(1000);

  BLEDevice::init("RepMotion_ESP32");
  BLEServer *pServer = BLEDevice::createServer();
  pServer->setCallbacks(new MyServerCallbacks());
  BLEService *pService = pServer->createService(SERVICE_UUID);
  pCharacteristic = pService->createCharacteristic(
    CHARACTERISTIC_UUID,
    BLECharacteristic::PROPERTY_NOTIFY);
  pCharacteristic->addDescriptor(new BLE2902());
  pService->start();
  BLEAdvertising *pAdvertising = BLEDevice::getAdvertising();
  pAdvertising->start();
  Serial.println("BLE ready, advertising...");








  delay(10000);
  Wire.begin(SDA_PIN, SCL_PIN);
  Wire.setClock(400000);

  Serial.println("Boot");

  // Check WHO_AM_I
  uint8_t who = 0;
  if (!readRegs(WHO_AM_I_REG, &who, 1)) {
    Serial.println("Failed to read WHO_AM_I");
    while (true) delay(1000);
  }

  Serial.print("WHO_AM_I = 0x");
  Serial.println(who, HEX);

  // MPU9250 WHO_AM_I is commonly 0x71
  // MPU6500 often 0x70 depending on module/chip variant
  if (who != 0x71 && who != 0x70) {
    Serial.println("Unexpected WHO_AM_I. Check sensor / wiring / address.");
  }

  // Wake up device
  writeReg(PWR_MGMT_1, 0x00);
  delay(100);

  // Optional clean config:
  // sample rate = gyro output rate / (1 + SMPLRT_DIV)
  writeReg(SMPLRT_DIV, 0x04);     // moderate rate
  writeReg(CONFIG_REG, 0x03);     // DLPF
  writeReg(GYRO_CONFIG, 0x00);    // ±250 dps
  writeReg(ACCEL_CONFIG, 0x00);   // ±2g
  writeReg(ACCEL_CONFIG2, 0x03);  // accel DLPF

  Serial.println("MPU9250 initialized");
  calibrateGyro();
  Serial.println("t_ms,ax_g,ay_g,az_g,gx_dps,gy_dps,gz_dps,acc_mag,gyro_mag,acc_mag_f,gyro_mag_f,is_still,roll_deg,pitch_deg,up_down,left_right,motion_state,temp_c");
}

void loop() {
  uint8_t raw[14];

  if (!readRegs(ACCEL_XOUT_H, raw, 14)) {
    Serial.println("Read failed");
    delay(200);
    return;
  }

  int16_t ax_raw = makeInt16(raw[0], raw[1]);
  int16_t ay_raw = makeInt16(raw[2], raw[3]);
  int16_t az_raw = makeInt16(raw[4], raw[5]);
  int16_t temp_raw = makeInt16(raw[6], raw[7]);
  int16_t gx_raw = makeInt16(raw[8], raw[9]);
  int16_t gy_raw = makeInt16(raw[10], raw[11]);
  int16_t gz_raw = makeInt16(raw[12], raw[13]);

  // Default full-scale config above:
  // accel ±2g => 16384 LSB/g
  // gyro ±250 dps => 131 LSB/(deg/s)
  float ax_g = ax_raw / 16384.0f;
  float ay_g = ay_raw / 16384.0f;
  float az_g = az_raw / 16384.0f;

  float gx_dps = (gx_raw / 131.0f) - gx_bias;
  float gy_dps = (gy_raw / 131.0f) - gy_bias;
  float gz_dps = (gz_raw / 131.0f) - gz_bias;

  float temp_c = (temp_raw / 333.87f) + 21.0f;

  float acc_mag = sqrt(ax_g * ax_g + ay_g * ay_g + az_g * az_g);
  float gyro_mag = sqrt(gx_dps * gx_dps + gy_dps * gy_dps + gz_dps * gz_dps);
  acc_mag_f = alpha * acc_mag + (1 - alpha) * acc_mag_f;
  gyro_mag_f = alpha * gyro_mag + (1 - alpha) * gyro_mag_f;

  bool is_still = (gyro_mag_f < 0.5) && (fabs(acc_mag_f - 1.0) < 0.15);

  float roll_deg = atan2(ay_g, az_g) * 180.0f / PI;
  float pitch_deg = atan2(-ax_g, sqrt(ay_g * ay_g + az_g * az_g)) * 180.0f / PI;

  const char *up_down_label = "flat";
  if (pitch_deg > tilt_thresh_deg) {
    up_down_label = "up";
  } else if (pitch_deg < -tilt_thresh_deg) {
    up_down_label = "down";
  }

  const char *left_right_label = "center";
  if (roll_deg > tilt_thresh_deg) {
    left_right_label = "right";
  } else if (roll_deg < -tilt_thresh_deg) {
    left_right_label = "left";
  }

  const char *motion_state = "idle";
  if (!is_still) {
    if (gy_dps > 5.0) {
      motion_state = "moving_up";
    } else if (gy_dps < -5.0) {
      motion_state = "moving_down";
    } else {
      motion_state = "moving_unclear";
    }
  }



  Serial.print(millis());
  Serial.print(",");

  Serial.print(ax_g, 6);
  Serial.print(",");
  Serial.print(ay_g, 6);
  Serial.print(",");
  Serial.print(az_g, 6);
  Serial.print(",");

  Serial.print(gx_dps, 6);
  Serial.print(",");
  Serial.print(gy_dps, 6);
  Serial.print(",");
  Serial.print(gz_dps, 6);
  Serial.print(",");

  Serial.print(acc_mag, 6);
  Serial.print(",");
  Serial.print(gyro_mag, 6);
  Serial.print(",");

  Serial.print(acc_mag_f, 6);
  Serial.print(",");
  Serial.print(gyro_mag_f, 6);
  Serial.print(",");

  Serial.print(is_still ? 1 : 0);
  Serial.print(",");

  Serial.print(roll_deg, 2);
  Serial.print(",");
  Serial.print(pitch_deg, 2);
  Serial.print(",");

  Serial.print(up_down_label);
  Serial.print(",");
  Serial.print(left_right_label);
  Serial.print(",");
  Serial.print(motion_state);
  Serial.print(",");

  Serial.println(temp_c, 2);



  if (deviceConnected && millis() - lastBleMs >= bleIntervalMs) {
    lastBleMs = millis();

    char msg[64];
    snprintf(
      msg,
      sizeof(msg),
      "%lu,%.3f,%.3f,%d",
      millis(),
      acc_mag_f,
      gyro_mag_f,
      is_still ? 1 : 0
    );

    pCharacteristic->setValue((uint8_t *)msg, strlen(msg));
    pCharacteristic->notify();

    Serial.print("BLE notify: ");
    Serial.println(msg);
  }

  delay(100);
}
